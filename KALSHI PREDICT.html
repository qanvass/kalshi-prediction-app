Ready-to-use ES6 Class. Automatically handles Kalshi's token bucket limits by intercepting `429` status codes and implementing mathematical exponential backoff. 

class KalshiClient {
    constructor(baseUrl = 'https://api.kalshi.com/trade-api/v2') {
        this.baseUrl = baseUrl;
    }
    async request(path, options = {}, retries = 5, delay = 1000) {
        try {
            const res = await fetch(`${this.baseUrl}${path}`, options);
            if (res.status === 429 && retries > 0) {
                await new Promise(r => setTimeout(r, delay));
                return this.request(path, options, retries - 1, delay * 2);
            }
            if (!res.ok) throw new Error(`API Error: ${res.statusText}`);
            return await res.json();
        } catch (e) {
            if (retries > 0) {
                await new Promise(r => setTimeout(r, delay));
                return this.request(path, options, retries - 1, delay * 2);
            }
            throw e;
        }
    }
    getMarkets() { return this.request('/markets'); }
    getOrderBook(ticker) { return this.request(`/markets/${ticker}/orderbook`); }
}

import time, requests

class KalshiV2Client:
    def __init__(self, base_url="https://api.kalshi.com/trade-api/v2"):
        self.base_url = base_url

    def execute(self, method, endpoint, retries=5, delay=1, **kwargs):
        for attempt in range(retries):
            try:
                r = requests.request(method, f"{self.base_url}{endpoint}", **kwargs)
                if r.status_code == 429:
                    time.sleep(delay)
                    delay *= 2
                    continue
                r.raise_for_status()
                return r.json()
            except requests.exceptions.RequestException as e:
                if attempt == retries - 1: raise e
                time.sleep(delay)
                delay *= 2

    def list_markets(self):
        return self.execute("GET", "/markets")
